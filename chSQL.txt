CREATE TABLE CUSTTOMER (
  Customer_id PRIMARY KEY VARCHAR2(20),
  Customer_Name NOT NULL VARCHAR2(20),
  Customer_Tel NUMBER
);

CREATE TABLE PRODUCT (
  Product_id PRIMARY KEY VARCHAR2(20),
  Product_Name NOT NULL VARCHAR2(20),
  Price NUMBER
);
ALTER TABLE PRODUCT ADD CONSTRAINT Category VARCHAR2(20);

CREATE TABLE ORDERS (
  Customer_id FOREIGN KEY VARCHAR2(20),
  Product_id FOREIGN KEY VARCHAR2(20),
  Quantity NUMBER,
  Total_amount NUMBER
);
ALTER TABLE ORDERS ADD CONSTRAINT OrderDate DATE SYSDATE();
